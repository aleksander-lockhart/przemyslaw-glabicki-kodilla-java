SELECT I.SUMMARY AS TASK_NAME, IL.NAME AS TASK_GROUP_NAME
FROM ISSUES I, ISSUESLISTS IL
WHERE IL.ID = I.ISSUESLIST_ID;

SELECT U.ID AS USER_ID, I.SUMMARY AS TASK_NAME
FROM ISSUES I, USERS U
WHERE I.ISSUESLIST_ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS HOW_MANY_TASKS
FROM ISSUES I, USERS U
WHERE I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 1
ORDER BY HOW_MANY_TASKS;
